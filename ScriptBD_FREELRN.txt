-- Creacion tabla TIPOS

CREATE TABLE TIPOS(
ID_TIPOS INT NOT NULL,
NOMBRE_TIPOS VARCHAR(20) NOT NULL,
PRIMARY KEY (ID_TIPOS)
);

-- Creacion tabla USUARIOS

CREATE TABLE USUARIOS(
USERNAME VARCHAR(20) NOT NULL, 
CONTRASEÑA VARCHAR(30) NOT NULL,
NOMBRE_USU VARCHAR(40) NOT NULL,
APELLIDO_USU VARCHAR(40) NOT NULL, 
CORREO_USU VARCHAR(50),
ID_TIPOS INT NOT NULL,
PRIMARY KEY (USERNAME),
FOREIGN KEY (ID_TIPOS) REFERENCES TIPOS(ID_TIPOS)
); 

-- Creacion tabla CURSOS 

CREATE TABLE CURSOS( 
ID_CURSOS INT NOT NULL AUTO_INCREMENT,
NOMBRE_CURSOS VARCHAR(100) NOT NULL,
ENLACE_CURSOS VARCHAR(100) NOT NULL, 
PRIMARY KEY (ID_CURSOS)
);

-- Creacion tabla USUARIOS_CURSOS

CREATE TABLE USUARIOS_CURSOS(
ID_CURSOS INT NOT NULL AUTO_INCREMENT,
USERNAME VARCHAR(20) NOT NULL, 
FOREIGN KEY (ID_CURSOS) REFERENCES CURSOS(ID_TIPOS),
FOREIGN KEY (USERNAME) REFERENCES USUARIOS(USERNAME)
);

-- Creacion tabla NOTICIAS

CREATE TABLE NOTICIAS(
ID_NOTICIAS INT NOT NULL AUTO_INCREMENT, 
INFO_NOTICIAS VARCHAR(100) NOT NULL,
PRIMARY KEY (ID_NOTICIAS)
);
